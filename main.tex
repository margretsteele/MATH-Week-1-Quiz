\section{What is the rank of matrix A?}
\[
  A=
\left[ {\begin{array}{cccc}
   0 & 0 & 0 & 0\\
   7 & 9 & 5 & 0\\
   1 & 0 & 0 & 0\\
   4 & 7 & 3 & 0
  \end{array} } \right]
\]

\begin{oneparchoices}
 \choice 0 
 \choice 1
 \choice 2
 \choice 3
\end{oneparchoices}
